    28  DifferentialDriveController(DifferentialDriveControllerParameters &parameters,

    31  parameters_(parameters), left_motor_(left_motor), right_motor_(right_motor)

    38  int set_polar_velocity(cogip_defs::Polar &command) override
    39  {
    40  // Compute wheel speed in mm/period from polar speed
    41  const double left_wheel_speed_mm_per_period = command.distance() - (command.angle() * parameters_.track_width_mm() / 2);
    42  const double right_wheel_speed_mm_per_period = command.distance() + (command.angle() * parameters_.track_width_mm() / 2);
    43 
    44  // Compute wheel speed in mm/s and rad/s from mm/period and rad/period speeds
    45  const double left_wheel_speed_mm_per_s = left_wheel_speed_mm_per_period * 1000 / parameters_.loop_period_ms();
    46  const double right_wheel_speed_mm_per_s = right_wheel_speed_mm_per_period * 1000/ parameters_.loop_period_ms();
    47 
    48  // Compute motor speed in percent using the motor constant.
    49  // The motor constant allow convert a speed in mm/s into a speed ratio (% of nominal motor voltage).
    50  const double left_motor_speed_percent = left_wheel_speed_mm_per_s / (etl::math::pi * parameters_.left_wheel_diameter_mm()) * parameters_.left_motor_constant();
    51  const double right_motor_speed_percent = right_wheel_speed_mm_per_s / (etl::math::pi * parameters_.right_wheel_diameter_mm()) * parameters_.right_motor_constant();
    52 
    53  // Apply motor speed
    54  left_motor_.speed((int)etl::clamp(left_motor_speed_percent, -parameters_.max_speed_percentage(), parameters_.max_speed_percentage()));
    55  right_motor_.speed((int)etl::clamp(right_motor_speed_percent, -parameters_.max_speed_percentage(), parameters_.max_speed_percentage()));
    56 
    57  return 0;
    58  }
    59 
    60 private:
    61  DifferentialDriveControllerParameters &parameters_;
    62 
    63  cogip::motor::MotorInterface &left_motor_;
    64  cogip::motor::MotorInterface &right_motor_;
    65 };
